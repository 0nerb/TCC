CREATE TABLE dim_acao (
    acao_key INTEGER PRIMARY KEY,
    ticker VARCHAR(20),
    nome VARCHAR(255),
    pais VARCHAR(100),
    setor VARCHAR(100),
    industria VARCHAR(100),
    data_coleta_dados DATE,
    data_listagem DATE
);

CREATE TABLE dim_tempo (
    data_key INTEGER PRIMARY KEY,
    data_completa DATE,
    ano INTEGER,
    mes INTEGER,
    dia INTEGER,
    trimestre INTEGER,
    nome_mes VARCHAR(20),
    dia_da_semana VARCHAR(20)
);

CREATE TABLE dim_tipoindicador (
    indicador_key INTEGER PRIMARY KEY,
    nome_indicador VARCHAR(100),
    descricao_indicador VARCHAR(255),
    formula_de_calculo VARCHAR(255)
);

CREATE TABLE fato_infoacaodiario (
    fk_dim_tempo_data_key INTEGER REFERENCES dim_tempo(data_key),
    fk_dim_acao_acao_key INTEGER REFERENCES dim_acao(acao_key),
    preco_fechamento DOUBLE PRECISION,
    volume_negociado DOUBLE PRECISION,
    rsi_14 DOUBLE PRECISION,
    PRIMARY KEY (fk_dim_tempo_data_key, fk_dim_acao_acao_key)
);

CREATE TABLE fato_indicadores (
    fk_dim_acao_acao_key INTEGER REFERENCES dim_acao(acao_key),
    fk_dim_tempo_data_key INTEGER REFERENCES dim_tempo(data_key),
    fk_dim_tipoindicador_indicador_key INTEGER REFERENCES dim_tipoindicador(indicador_key),
    valor_indicador DOUBLE PRECISION,
    PRIMARY KEY (fk_dim_acao_acao_key, fk_dim_tempo_data_key, fk_dim_tipoindicador_indicador_key)
);
